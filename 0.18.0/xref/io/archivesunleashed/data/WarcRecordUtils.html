16  package io.archivesunleashed.data;
17  
18  import java.io.BufferedInputStream;
19  import java.io.ByteArrayInputStream;
20  import java.io.ByteArrayOutputStream;
21  import java.io.DataOutputStream;
22  import java.io.IOException;
23  import java.io.InputStream;
24  import java.util.Map;
25  import java.util.regex.Matcher;
26  import java.util.regex.Pattern;
27  import org.apache.commons.httpclient.HttpParser;
28  import org.apache.commons.io.IOUtils;
29  import org.apache.commons.io.input.BoundedInputStream;
30  import org.apache.log4j.Logger;
31  import org.archive.format.warc.WARCConstants;
32  import org.archive.io.warc.WARCReader;
33  import org.archive.io.warc.WARCReaderFactory;
34  import org.archive.io.warc.WARCRecord;
35  
36  /**
37   * Utilities for working with {@code WARCRecord}s (from archive.org APIs).
38   */
39  public final class WarcRecordUtils implements WARCConstants {
40  
41    /**
42     * Utility classes should not have a public or default constructor.
43     */
44    private WarcRecordUtils() {
45    }
46  
47    /**
48    * Setup logger.
49    */
50    private static final Logger LOG = Logger.getLogger(WarcRecordUtils.class);
51  
52    /**
53     * Converts raw bytes into an {@code WARCRecord}.
54     *
55     * @param bytes raw bytes
56     * @return parsed {@code WARCRecord}
57     * @throws IOException if there is an issue
58     */
59    public static WARCRecord fromBytes(final byte[] bytes) throws IOException {
60      WARCReader reader = (WARCReader) WARCReaderFactory.get("",
61          new BufferedInputStream(new ByteArrayInputStream(bytes)), false);
62      return (WARCRecord) reader.get();
63    }
64  
65    /**
66     * Converts WARC record into raw bytes.
67     *
68     * @param record conents of WARC response record
69     * @return raw contents
70     * @throws IOException if there is an issue
71     */
72    public static byte[] toBytes(final WARCRecord record) throws IOException {
73      ByteArrayOutputStream baos = new ByteArrayOutputStream();
74      DataOutputStream dout = new DataOutputStream(baos);
75  
76      dout.write("WARC/0.17\n".getBytes());
77      for (Map.Entry<String, Object> entry : record.getHeader()
78              .getHeaderFields().entrySet()) {
79        dout.write((entry.getKey() + ": " + entry.getValue().toString() + "\n")
80                .getBytes());
81      }
82      dout.write("\n".getBytes());
83      record.dump(dout);
84  
85      return baos.toByteArray();
86    }
87  
88    /**
89     * Extracts the MIME type of WARC response records.
90     * "WARC-Type" is "response".
91     * Note that this is different from the "Content-Type" in the WARC header.
92     *
93     * @param contents raw contents of the WARC response record
94     * @return MIME type
95     */
96    public static String getWarcResponseMimeType(final byte[] contents) {
97      // This is a somewhat janky way to get the MIME type of the response.
98      // Moreover, this simple regex is not compliant with the specification.
99      // See: https://www.w3.org/Protocols/rfc1341/4_Content-Type.html
100     // It would be much better to parse all headers using an external library:
101     //   org.apache.commons.httpclient.HeaderElement
102     // Note that this is different from the "Content-Type" in the WARC header.
103     Pattern pattern = Pattern.compile("Content-Type: ([^\\s;]+) *(;.*)?",
104             Pattern.CASE_INSENSITIVE);
105     Matcher matcher = pattern.matcher(new String(contents));
106     if (matcher.find()) {
107       return matcher.group(1).replaceAll(";$", "");
108     }
109 
110     return null;
111   }
112 
113   /**
114    * Extracts raw contents from a {@code WARCRecord} (including HTTP headers).
115    *
116    * @param record the {@code WARCRecord}
117    * @return raw contents
118    * @throws IOException if there is an issue
119    */
120   public static byte[] getContent(final WARCRecord record) throws IOException {
121     int len = (int) record.getHeader().getContentLength();
122 
123     // If we have a corrupt record, quit and move on.
124     if (len < 0) {
125       return new byte[0];
126     }
127 
128     try {
129       return copyToByteArray(record, len, true);
130     } catch (Exception e) {
131       // Catch exceptions related to any corrupt archive files.
132       return new byte[0];
133     }
134   }
135 
136   /**
137    * Extracts contents of the body from a {@code WARCRecord}.
138    * Excludes HTTP headers.
139    *
140    * @param record the {@code WARCRecord}
141    * @return contents of the body
142    * @throws IOException if there is an issue
143    */
144   public static byte[] getBodyContent(final WARCRecord record)
145       throws IOException {
146     ByteArrayOutputStream baos = new ByteArrayOutputStream();
147     String line = HttpParser.readLine(record, WARC_HEADER_ENCODING);
148     if (line == null) {
149       return null;
150     }
151 
152     // Just using parseHeaders to move down input stream to body
153     HttpParser.parseHeaders(record, WARC_HEADER_ENCODING);
154     record.dump(baos);
155     return baos.toByteArray();
156   }
157 
158   /**
159    * Copies contents to a byte array.
160    *
161    * @param is raw input stream
162    * @param recordLength length of a record
163    * @param enforceLength enforce the length
164    * @return rawContents of body
165    * @throws IOException if there is an issue
166    */
167   private static byte[] copyToByteArray(final InputStream is,
168           final int recordLength, final boolean enforceLength)
169       throws IOException {
170 
171     BoundedInputStream bis = new BoundedInputStream(is, recordLength);
172     byte[] rawContents = IOUtils.toByteArray(bis);
173     if (enforceLength && rawContents.length != recordLength) {
174       LOG.error("Read " + rawContents.length + " bytes but expected "
175               + recordLength + " bytes. Continuing...");
176     }
177     return rawContents;
178   }
179 }
